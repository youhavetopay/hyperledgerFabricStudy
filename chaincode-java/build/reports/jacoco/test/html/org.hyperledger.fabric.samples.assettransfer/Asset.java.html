package org.hyperledger.fabric.samples.assettransfer;

import java.util.Objects;

import org.hyperledger.fabric.contract.annotation.DataType;
import org.hyperledger.fabric.contract.annotation.Property;

import com.owlike.genson.annotation.JsonProperty;

@DataType()
public final class Asset {

    @Property()
    private final String assetID;

    @Property()
    private final String userId;

    @Property()
    private final String pw;

    @Property()
    private final String userName;

    public String getAssetID() {
        return assetID;
    }

    public String getUserId() {
        return userId;
    }

    public String getPw() {
        return pw;
    }

    public String getUserName() {
        return userName;
    }

    public Asset(@JsonProperty("assetID") final String assetID, @JsonProperty("userId") final String userId,
           @JsonProperty("pw") final String pw, @JsonProperty("userName") final String userName) {
        this.assetID = assetID;
        this.userId = userId;
        this.pw = pw;
        this.userName = userName;
    }

    @Override
    public boolean equals(final Object obj) {
        if (this == obj) {
            return true;
        }

        if ((obj == null) || (getClass() != obj.getClass())) {
            return false;
        }

        Asset other = (Asset) obj;

        return Objects.deepEquals(
                new String[] {getAssetID(), getUserId(), getPw(), getUserName()},
                new String[] {other.getAssetID(), other.getUserId(), other.getPw(), other.getUserName()});
    }

    @Override
    public int hashCode() {
        return Objects.hash(getAssetID(), getUserId(), getPw(), getUserName());
    }

    @Override
    public String toString() {
        return this.getClass().getSimpleName() + "@" + Integer.toHexString(hashCode()) + " [assetID=" + assetID + ", userId="
                + userId + ", pw=" + pw + ", userName=" + userName + "]";
    }
}
